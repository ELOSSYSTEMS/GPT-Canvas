🐒 Tampermonkey Script – ELoS Prompt Autocomplete Simulator
// ==UserScript==
// @name         ELoS Prompt Autocomplete Simulator
// @namespace    http://tampermonkey.net/
// @version      0.1
// @description  Simulates prompt suggestion overlay from a local library
// @author       Allon
// @match        *://*/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    const PROMPT_LIBRARY = [
        {
            title: "🧠 Self-Interrogation Loop",
            text: "What question am I avoiding most? Ask it directly, answer it without justification, then ask the next layer deeper.",
            keywords: ["improve", "self", "question", "growth"]
        },
        {
            title: "⏳ Time Scarcity Reframe",
            text: "Assume I am not busy—I am misaligned. Rebuild my schedule from scratch based on what deserves time, not what demands it.",
            keywords: ["time", "schedule", "improve"]
        },
        {
            title: "📊 Strategic Burn Audit",
            text: "Run a burn audit—not just cash, but attention, energy, and time. Where am I overspending strategically with low ROI?",
            keywords: ["burn", "energy", "improve"]
        },
        {
            title: "⚙️ Habit Loop Dissection",
            text: "Choose one recurring habit or pattern. Break it into trigger, craving, response, and reward. Rewire it surgically.",
            keywords: ["habit", "improve", "routine"]
        }
    ];

    let overlay;
    let activeTarget;
    let timeout;
    let lastText = "";

    function createOverlay() {
        overlay = document.createElement("div");
        overlay.id = "elos-overlay";
        overlay.style.position = "absolute";
        overlay.style.background = "#fff";
        overlay.style.border = "1px solid #ccc";
        overlay.style.borderRadius = "6px";
        overlay.style.boxShadow = "0 4px 12px rgba(0,0,0,0.15)";
        overlay.style.zIndex = 999999;
        overlay.style.fontSize = "14px";
        overlay.style.padding = "4px 0";
        overlay.style.display = "none";
        overlay.style.minWidth = "300px";
        overlay.style.maxWidth = "400px";
        document.body.appendChild(overlay);
    }

    function showOverlay(suggestions, rect) {
        overlay.innerHTML = "";
        suggestions.forEach(prompt => {
            const el = document.createElement("div");
            el.style.padding = "6px 10px";
            el.style.cursor = "pointer";
            el.style.borderBottom = "1px solid #eee";
            el.innerHTML = `<strong>${prompt.title}</strong><br><span style="color:#555">${prompt.text}</span>`;
            el.onclick = () => {
                if (activeTarget) {
                    insertPrompt(prompt.text);
                    hideOverlay();
                }
            };
            overlay.appendChild(el);
        });
        overlay.style.top = `${rect.bottom + window.scrollY}px`;
        overlay.style.left = `${rect.left + window.scrollX}px`;
        overlay.style.display = "block";
    }

    function hideOverlay() {
        overlay.style.display = "none";
    }

    function insertPrompt(text) {
        if (activeTarget.tagName === "TEXTAREA" || activeTarget.tagName === "INPUT") {
            const start = activeTarget.selectionStart;
            const value = activeTarget.value;
            activeTarget.value = value.slice(0, start) + text + value.slice(start);
        } else if (activeTarget.isContentEditable) {
            const sel = window.getSelection();
            const range = sel.getRangeAt(0);
            range.deleteContents();
            range.insertNode(document.createTextNode(text));
        }
    }

    function matchPrompts(input) {
        const tokens = input.toLowerCase().split(/\s+/);
        return PROMPT_LIBRARY.filter(p =>
            p.keywords.some(k => tokens.includes(k))
        );
    }

    function attachListeners(el) {
        el.addEventListener("input", () => {
            clearTimeout(timeout);
            timeout = setTimeout(() => {
                const text = el.value || el.innerText;
                if (text !== lastText) {
                    lastText = text;
                    const suggestions = matchPrompts(text);
                    if (suggestions.length) {
                        const rect = el.getBoundingClientRect();
                        activeTarget = el;
                        showOverlay(suggestions, rect);
                    } else {
                        hideOverlay();
                    }
                }
            }, 300);
        });
        el.addEventListener("blur", () => setTimeout(hideOverlay, 200));
    }

    function observe() {
        const observer = new MutationObserver(() => {
            document.querySelectorAll("textarea:not([data-elos-attached]), [contenteditable]:not([data-elos-attached])").forEach(el => {
                el.setAttribute("data-elos-attached", "true");
                attachListeners(el);
            });
        });
        observer.observe(document.body, { childList: true, subtree: true });
    }

    createOverlay();
    observe();
})();

// @namespace    http://tampermonkey.net/

// @match        https://chat.openai.com/*
// @grant        none

(function () {
  "use strict";

  /* ---------- tiny prompt library ---------- */
  const LIB = [
    { t: "⏳ Time Scarcity Reframe",
      p: "Assume I am not busy—I am misaligned. Rebuild my schedule from scratch based on what deserves time, not what demands it.",
      k: ["time","schedule","improve"] },
    { t: "🧠 Self-Interrogation Loop",
      p: "What question am I avoiding most? Ask it directly, answer it without justification, then ask the next layer deeper.",
      k: ["improve","question","growth"] }
  ];

  /* ---------- overlay ---------- */
  let overlay = document.createElement("div");
  Object.assign(overlay.style, {
    position:"absolute",background:"#fff",border:"1px solid #bbb",
    borderRadius:"6px",boxShadow:"0 4px 12px rgba(0,0,0,.15)",
    fontSize:"13px",maxWidth:"430px",zIndex:2e9,display:"none"
  });
  document.body.appendChild(overlay);
  const hide = ()=>overlay.style.display="none";

  function show(list, rect){
    overlay.innerHTML="";
    list.forEach(o=>{
      let d=document.createElement("div");
      d.innerHTML=<strong>${o.t}</strong><br><span>${o.p}</span>;
      Object.assign(d.style,{padding:"6px 8px",cursor:"pointer",borderBottom:"1px solid #eee"});
      d.onclick=()=>{insert(o.p); hide();};
      overlay.appendChild(d);
    });
    overlay.style.top=${rect.bottom+scrollY}px;
    overlay.style.left=${rect.left+scrollX}px;
    overlay.style.display="block";
  }

  /* ---------- matching ---------- */
  const match = txt=>{
    const tok=txt.toLowerCase().split(/\s+/);
    return LIB.filter(o=>o.k.some(k=>tok.includes(k)));
  };

  /* ---------- insertion ---------- */
  let field=null;
  const insert = text=>{
    if(!field) return;
    document.execCommand("insertText",false," "+text+" ");
  };

  /* ---------- attach logic ---------- */
  const attach = node=>{
    if(node.dataset.elosAttached) return;
    node.dataset.elosAttached="1";
    console.log("[ELoS] attached");
    let timer=0;
    node.addEventListener("input",()=>{clearTimeout(timer);
      timer=setTimeout(()=>evaluate(node),300);});
    node.addEventListener("blur",hide);
    document.addEventListener("keydown",e=>{
      if(e.altKey && e.key==="p"){evaluate(node);}  // manual trigger
    });
  };

  const evaluate = n=>{
    field=n;
    const val=n.innerText;
    const hits=match(val);
    if(hits.length) show(hits,n.getBoundingClientRect());
    else hide();
  };

  /* ---------- observer to re-bind ---------- */

    const composerSelector='div#prompt-textarea[contenteditable="true"]';
    const bindCurrent=()=>{const el=document.querySelector(composerSelector); if(el) attach(el);};
    bindCurrent();
    new MutationObserver(bindCurrent).observe(document.body,{childList:true,subtree:true});
  };

  watch();
})();
